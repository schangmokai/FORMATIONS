# Installer openssh-server et libxml2-utils

    apt install -y openssh-server libxml2-utils && \
    mkdir -p /var/run/sshd

    echo 'mokai:mokai' | chpasswd

    echo 'PasswordAuthentication yes' >> /etc/ssh/sshd_config
